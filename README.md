IMPORTANT COMMANDS
------------------

- Check versions

    node -v \
    npm -v

- Global installation 

    npm i typescript -g \
    npm i ts-node -g \
    npm i nodemon -g

- Prepare project 

    tsc --init \
    npm init -y \
    npm i express cors dotenv multer mongoose \
    npm i @types/express @types/cors @types/dotenv @types/multer @types/mongoose -D
